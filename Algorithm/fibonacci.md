```javascript
function fibonacci(n) {
  // TODO: 여기에 코드를 작성합니다.

  // n번째 피보나치 항을 여러 번 계산한다고 해서 그 값이 달라지지 않는다.
  // 즉, 한번 구한 항을 기록(memoization) 해 둔 다음, 그 값이 필요할 때 꺼내쓸 수 있다면 굉장히 효율적일 것이다.

  // 1. 문제를 부분 문제로 나눈다.
  // 2. 가장 작은 문제를 해를 구한 뒤, 테이블에 저장한다.
  // 3. 테이블에 저장되어있는 데이터로 전체의 문제의 해를 구한다.

  // 일단 초기 배열이 [0, 1]에서 시작하여 배열의 요소를 누적해 나가는 방법
  // 그리고 이미 구해놓은 것은 배열의 요소로 저장해놓기..!!! 그래야 런타임이 초과되지 않는다

  // -solution-
  // [0,1]은 고정
  // 함수선언, -> let fibo = (n) =>
  // 이미있는건 그대로리턴
  // 없는건 저장 변수에 저장 memo[n] = ~
  // 재귀함수사용 (fibo-1) + (fibo-2)
  //fibo[n]리턴

  let memo = [0,1]
  let fibo = (n) => {
    if(memo[n] !== undefined){
      return memo[n]
    }
    memo[n] = fibo(n - 1) + fibo(n - 2)
    return memo[n]
  }
  return fibo(n)

}
```
